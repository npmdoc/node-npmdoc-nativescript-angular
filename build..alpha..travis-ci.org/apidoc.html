    function findProjectDir() {
    var candidateDir = path.join(__dirname, "..");
    while (true) {
        var oldCandidateDir = candidateDir;
        candidateDir = path.dirname(candidateDir);

        if (fs.existsSync(packageJsonFile)) {
            return candidateDir;
        }
        if (oldCandidateDir === candidateDir) {
            return;
        }
    }
}
